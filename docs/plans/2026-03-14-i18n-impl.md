# i18n Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add English/Korean i18n support using `next-intl` with path-based routing (`/ko/*`), browser language auto-detection, and no prefix for the default English locale.

**Architecture:** Install `next-intl`, create routing/request/navigation config files, restructure `app/` into `app/[locale]/`, combine i18n middleware with existing Supabase session middleware, extract ~300 hardcoded strings into `messages/en.json` and `messages/ko.json`.

**Tech Stack:** `next-intl`, Next.js 15 App Router, TypeScript

import { routing } from "./routing";

export default getRequestConfig(async ({ requestLocale }) => {
  const requested = await requestLocale;
  const locale = hasLocale(routing.locales, requested)
    ? requested
    : routing.defaultLocale;

  return {
    locale,
    messages: (await import(`../messages/${locale}.json`)).default,
  };
});

### Task 4: Update middleware

Combine next-intl locale routing with Supabase session management.

**Files:**
- Modify: `middleware.ts`
- Modify: `lib/supabase/middleware.ts`

**Step 1: Update `middleware.ts`**

```typescript
import { type NextRequest } from "next/server";
import createMiddleware from "next-intl/middleware";
import { routing } from "@/i18n/routing";
import { updateSession } from "@/lib/supabase/middleware";

const handleI18nRouting = createMiddleware(routing);

export async function middleware(request: NextRequest) {
  // Step 1: Handle locale routing (redirects, rewrites)
  const response = handleI18nRouting(request);

  // Step 2: Update Supabase session on the same response object
  await updateSession(request, response);

  return response;
}

export const config = {
  matcher: [
    "/((?!api|_next/static|_next/image|favicon.ico|logo|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)",
  ],
};
```

**Step 2: Update `lib/supabase/middleware.ts`**

Change `updateSession` to accept an existing response instead of creating a new one:

```typescript
import { createServerClient } from "@supabase/ssr";
import { type NextRequest, type NextResponse } from "next/server";

export async function updateSession(request: NextRequest, response: NextResponse) {
  // Skip auth call when no Supabase auth cookies exist (anonymous users)
  const hasAuthCookie = request.cookies
    .getAll()
    .some((c) => c.name.startsWith("sb-") && c.name.includes("-auth-token"));
  if (!hasAuthCookie) return;

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_DEFAULT_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll();
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) => {
            response.cookies.set(name, value, options);
          });
        },
      },
    },
  );

  await supabase.auth.getUser();
}
```

Note: The function no longer returns a response — it mutates the one passed in. Check all callers of `updateSession` (only `middleware.ts`).

"use client";

import { usePathname, useRouter } from "@/i18n/navigation";
import { useLocale } from "next-intl";
import { Button } from "@/components/ui/button";

export function LocaleSwitcher() {
  const pathname = usePathname();
  const router = useRouter();
  const locale = useLocale();

  function handleSwitch() {
    const nextLocale = locale === "en" ? "ko" : "en";
    router.replace(pathname, { locale: nextLocale });
  }

  return (
    <Button variant="ghost" size="sm" onClick={handleSwitch} className="text-xs">
      {locale === "en" ? "한" : "EN"}
    </Button>
  );
}

## Reference: Key Patterns

### Server Component page pattern

```typescript
import { use } from "react";
import { setRequestLocale } from "next-intl/server";
import { useTranslations } from "next-intl";

type Props = { params: Promise<{ locale: string }> };

export default function SomePage({ params }: Props) {
  const { locale } = use(params);
  setRequestLocale(locale);
  const t = useTranslations("namespace");
  return <h1>{t("key")}</h1>;
}
```

### Async Server Component page pattern

```typescript
import { getTranslations, setRequestLocale } from "next-intl/server";

type Props = { params: Promise<{ locale: string }> };

export default async function SomePage({ params }: Props) {
  const { locale } = await params;
  setRequestLocale(locale);
  const t = await getTranslations("namespace");
  return <h1>{t("key")}</h1>;
}
```

### Client Component pattern

```typescript
"use client";
import { useTranslations } from "next-intl";

export function SomeComponent() {
  const t = useTranslations("namespace");
  return <button>{t("label")}</button>;
}
```

### Metadata pattern

```typescript
import { getTranslations } from "next-intl/server";

export async function generateMetadata({ params }: { params: Promise<{ locale: string }> }) {
  const { locale } = await params;
  const t = await getTranslations({ locale, namespace: "namespace" });
  return { title: t("pageTitle"), description: t("pageDescription") };
}
```

### Interpolation in translations

```json
{ "greeting": "Hello, {name}!" }
```
```typescript
t("greeting", { name: "John" })
```

### Pluralization

```json
{ "count": "{count, plural, one {# item} other {# items}}" }
```
```typescript
t("count", { count: 5 })
```
